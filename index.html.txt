<script>
const sequence = ["black","pink","red","grey","green","orange","blue"];

const labels = {
  black:"SVART",
  pink:"ROSA",
  red:"RØD",
  grey:"GRÅ",
  green:"GRØNN",
  orange:"ORANSJE",
  blue:"BLÅ"
};

function render() {
  const el = document.getElementById("bottom");
  el.innerHTML = "";

  const now = new Date();

  const clock = document.createElement("div");
  clock.className = "clock";
  clock.textContent = now.toLocaleTimeString("no-NO",{hour:"2-digit",minute:"2-digit"});
  el.appendChild(clock);

  const rows = document.createElement("div");
  rows.className = "rows";
  el.appendChild(rows);

  for (let i = 0; i < 5; i++) {
    const div = document.createElement("div");
    div.className = "row " + sequence[i] + (i===0?" first":"");
    div.innerHTML = `<span>${labels[sequence[i]]}</span><span>${Math.floor(Math.random()*60)} min</span>`;
    rows.appendChild(div);
  }
}

setInterval(render, 1000);
render();
</script>
